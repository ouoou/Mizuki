public int numberOfPairs(int[][] points) {  
    int ans = 0;  
    for (int i = 0; i < points.length; i++) {  
        int[] a = points[i];  
        for (int j = 0; j < points.length; j++) {  
            if (i == j) continue;  
            int[] b = points[j];  
            if (a[0] <= b[0] && a[1] >= b[1]) {  
                boolean ok = true;  
                for (int k = 0;  k < points.length; k++) {  
                    if (i == k || j == k) continue;  
                    int[] c = points[k];  
                    if (c[0] >= a[0] && c[1] <= a[1] && c[0] <= b[0] && c[1] >= b[1]) {  
                        ok = false;  
                        break;  
                    }  
                }  
                if (ok) ans++;  
            }  
        }  
    }  
    return ans;  
}

    }
