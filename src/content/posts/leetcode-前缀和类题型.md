        long sum = 0;
        for (int i = 0; i < nums.length; i++) {
            sum += nums[i];
        }

            }
        }

        int ans = 0;
        for (int j = n; j >= 0; j--) {
            while (!deque.isEmpty() && prefix[j] > prefix[deque.peek()]) {
                ans = Math.max(ans, j - deque.pop());
            }
        }
        return ans;
    }
}
