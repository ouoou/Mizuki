class Solution {
    public long bowlSubarrays(int[] nums) {
        int n = nums.length;
        int [] l = new int[n];
        int [] r = new int[n];
        Arrays.fill(l, -1);
        Arrays.fill(r, -1);

        Stack<Integer> stack = new Stack<>();
        for (int i = 0; i < nums.length; i++) {
            while (!stack.isEmpty() && nums[stack.peek()] < nums[i]) {
                stack.pop();
            }
            if (!stack.isEmpty()) {
                l[i] = stack.peek();
            }
            stack.push(i);
        }
        stack.clear();

        for (int i = n - 1; i >= 0; i--) {
            while (!stack.isEmpty() && nums[stack.peek()] < nums[i]) {
                stack.pop();
            }
            if (!stack.isEmpty()) {
                r[i] = stack.peek();
            }
            stack.push(i);
        }

        int ans = 0;
        for (int i = 0; i < n; i++) {
            if (l[i] != -1 && r[i] != -1 && r[i] - l[i] + 1 >= 3) {
                ans ++;
            }
        }
        return ans;
    }
}

        if (n == 0) {
            return 1;
        }
        String s = Long.toBinaryString(n);
        int l = 0;
        int r = s.length() - 1;
        while (l < r) {
            if (s.charAt(l) == s.charAt(r)) {
                l++;
                r--;
            } else {
                return 0;
            }
        }
        return 1;
    }
