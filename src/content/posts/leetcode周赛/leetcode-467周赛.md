class Solution {
    public boolean[] subsequenceSumAfterCapping(int[] nums, int k) {
        int n = nums.length;
        int[] arr = new int[n];
        boolean[] ans = new boolean[n];
        for (int x = 1; x <= n; x++) {
            for (int i = 0; i < n; i++) {
                arr[i] = Math.min(x, nums[i]);
            }
            
            if(judge(arr, k)) {
                ans[x - 1] = true;
            }
        }
        return ans;
    }
    public boolean judge(int[] arr, int k) {
        int n = arr.length;
        int total = 1 << n;
        for (int m = 1; m <= total; m++) {
            int sum = 0;
            for (int i = 0; i < n; i++) {
                if ((m & (1 << i)) != 0) {
                    sum += arr[i];
                }
            }
            if (sum == k) {
                return true;
            }
        }
        return false;
    }
}
